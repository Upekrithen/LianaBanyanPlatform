```sql
-- Core IP Ledger table (per-Ring-Bearer, local-first)
CREATE TABLE IF NOT EXISTS ip_ledger (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  ring_bearer_id  UUID NOT NULL,                  -- references peers(id) or members(id)
  entry_type      TEXT NOT NULL,                  -- 'config_set' | 'model_pull' | 'broadcast' | 'canon_seal'
  payload_hash    TEXT NOT NULL,                  -- SHA-256 or Blake3 of the entry payload
  payload_json    JSONB NOT NULL,                 -- full payload for audit
  ed25519_sig     TEXT,                           -- Ed25519 signature of payload_hash by ring_bearer
  stamp_seq       BIGINT NOT NULL DEFAULT 0,      -- monotonic sequence per ring_bearer
  stamped_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  merkle_node     TEXT,                           -- Merkle tree node hash for mesh replication
  replicated_at   TIMESTAMPTZ,                    -- null until mesh peer confirms receipt
  CONSTRAINT ip_ledger_ring_seq UNIQUE (ring_bearer_id, stamp_seq)
);

-- Merkle diff replication log (mesh-layer)
CREATE TABLE IF NOT EXISTS ip_ledger_merkle_diff (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  source_peer_id  UUID NOT NULL,
  target_peer_id  UUID NOT NULL,
  root_hash       TEXT NOT NULL,                  -- Merkle root at time of diff
  diff_payload    JSONB NOT NULL,                 -- entries the target is missing
  transmitted_at  TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  acked_at        TIMESTAMPTZ
);

-- Index for fast Ring Bearer lookups
CREATE INDEX IF NOT EXISTS ip_ledger_ring_bearer_idx
  ON ip_ledger (ring_bearer_id, stamped_at DESC);

-- Index for replication lag queries
CREATE INDEX IF NOT EXISTS ip_ledger_unreplicated_idx
  ON ip_ledger (stamped_at DESC)
  WHERE replicated_at IS NULL;

COMMIT;
```
